\Large\textbf{Слабая формула}\\
$$T\in Q=\{v\in H:v|_{Г_1\cup{Г_2}\cup{Г_3}\cup{Г_4}}=0\},$$
$$u\in V=[H(\Omega)]^d,\quad d=2$$
Уравнение упругости
$$\int_{\Omega}\sigma(u)\varepsilon(u)dx+\int_{\Omega}\beta{(grad(T-T_0),v)dx}=0, \quad \forall{v}\in{\widehat{V}},$$
Уравнение тепла
$$\int_{\Omega}C\frac{T-\check{T}}{\tau}qdx+\int_{\Omega}\beta{T_0}\frac{div\;u-div\;\check{u}}{\tau}qdx+$$
$$+\int_{\Omega}(kgrad\;T,grad\;q)dx=\int_{\Omega}fqdx,\quad\forall{q}\in{\widehat{Q}}$$
